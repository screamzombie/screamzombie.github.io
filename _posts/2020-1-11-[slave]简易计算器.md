### 简单计算器

```c++
#include <iostream>
#include <vector>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <string>
#include <windows.h>
#include <cassert>
#include <ctime>
using namespace std;

//  *全局变量区域
bool last_answer_to_use = false; //是否保留上次一次的结果作为运算。
string numbers = "";
string before_number = "";
string after_number = "";
char mid_symbel;
double last_answer; //上一个结果

void mysplit(string words) //分割数字
{
    int index = 0;
    for (int i = 0; i < words.size(); i++)
    {
        if (words[i] == '+' || words[i] == '-' || words[i] == '*' || words[i] == '/') //也就是找到分隔符了
        {
            index = i;
            break;
        }
    }
    before_number = words.substr(0, index); //前一个数字
    after_number = words.substr(index + 1); //后一个数字
    mid_symbel = words[index];              //符号占位
}

void getTime()
{
    time_t rawtime;
    struct tm *ptminfo;
    time(&rawtime);
    ptminfo = localtime(&rawtime);
    printf("现在时间是: %02d-%02d-%02d %02d:%02d:%02d\n",
           ptminfo->tm_year + 1900, ptminfo->tm_mon + 1, ptminfo->tm_mday,
           ptminfo->tm_hour, ptminfo->tm_min, ptminfo->tm_sec);
}

int getSymbol(string word)
{
    int count = 0;
    for (int i = 0; i < word.size(); i++)
    {
        if (word[i] == '+' || word[i] == '-' || word[i] == '*' || word[i] == '/')
        {
            count++;
        }
    }
    return count;
}

void display()
{
    cout << "欢迎使用简易计算器" << endl;
    getTime();
    string line;
    cout << "请输入算式:" << endl;
    cin >> line;
    int count = 0;
    if (line.find('+') == string::npos && line.find('-') == string::npos && line.find('*') == string::npos && line.find('/') == string::npos)
    {
        cout << "输入错误请重新输入" << endl;
        display();
    }
    else if (getSymbol(line) > 1)
    {
        cout << "运算符不能超过一个，请重新输入" << endl;
        display();
    }
    else
    {
        numbers = line;
    }
}

double singleStringToDouble(char ch)
{
    switch (ch)
    {
    case '0':
        return 0;
    case '1':
        return 1;
    case '2':
        return 2;
    case '3':
        return 3;
    case '4':
        return 4;
    case '5':
        return 5;
    case '6':
        return 6;
    case '7':
        return 7;
    case '8':
        return 8;
    case '9':
        return 9;
    default:
        cout << "错误!" << endl;
        break;
    }
}

double myStringToDouble(string num) // 字符串转double
{
    bool flag = false; //小数位还是整数位的标记 默认是整数
    double total = 0;
    int index = 0;
    for (int i = 0; i < num.size(); i++)
    {
        if (num[i] == '.') //说明遇到小数点了,剩下的应该进到小数了
        {
            index = i;
            flag = true;
            break;
        }
        else
        {
            total *= 10;
            total += singleStringToDouble(num[i]);
        }
    }
    index++;
    double zoom = 0.1; //放缩比例
    for (int i = index; i < num.size(); i++)
    {
        total += zoom * singleStringToDouble(num[i]);
        zoom *= 0.1;
    }
    return total;
}

double answer(string before_number, char mid_symbel, string after_number) //计算结果的函数
{
    double before_num = myStringToDouble(before_number);
    if (last_answer_to_use == true)
    {
        before_num = last_answer;
        last_answer_to_use = false;
    }
    double after_num = myStringToDouble(after_number);
    switch (mid_symbel)
    {
    case '+':
        return before_num + after_num;

    case '-':
        return before_num - after_num;

    case '*':
        return before_num * after_num;

    case '/':
        return before_num / after_num;
    default:
        cout << "error" << endl;
        break;
    }
}

void calculater() //计算函数
{
    cout << "请输入算式" << endl;
    mysplit(numbers);
    double ans = answer(before_number, mid_symbel, after_number);
    cout << "运算结果是" << ans << endl;
    last_answer = ans;
    cout << "是否保留运算结果，输入1保留，输入0退出" << endl;
    cin >> last_answer_to_use;
    while (last_answer_to_use)
    {
        string temp;
        cout << "请输入新的式子" << endl;
        cin >> temp;
        numbers = "0" + temp;
        mysplit(numbers);
        ans = answer(before_number, mid_symbel, after_number);
        cout << "运算结果是" << ans << endl;
        cout << "是否保留运算结果，输入1保留，输入0退出" << endl;
        cin >> last_answer_to_use;
        if (last_answer_to_use == true)
        {
            last_answer = ans;
        }
    }
    cout << "感谢使用！" << endl;
}

int main()
{
    display();
    calculater();
    return 0;
}
```
