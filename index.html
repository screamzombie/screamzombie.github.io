Makefile学习笔记
计算机技术 2024/5/19

一个例子

Compile run all cpp files of src

CC = clang++
CCFLAGS = -std=c++2b -Wall
TARGET = test
SRCS = $(wildcard ./src/*.cpp)
OBJS = $(patsubst %.cpp,%.o,$(SRCS))
all:$(TARGET)
    mv $(TARGET) ./bin
    mv $(OBJS) ./build
$(TARGET): $(OBJS)
    $(CC) $(CCFLAGS) -o $@ $^

%.o: %.cpp
    $(CC) $(CCFLAGS) -c $< -o $@
.PHONY:clean
clean:
    rm -f bin/* build/*%
